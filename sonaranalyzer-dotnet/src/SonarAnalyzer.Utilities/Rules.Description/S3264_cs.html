在任何地方都没有调用的事件是死代码，没有理由在源代码里继续保留它们。

不合规的代码示例

class UninvokedEventSample
{
    private event Action<object, EventArgs> Happened; //不合规

    public void RegisterEventHandler(Action<object, EventArgs> handler)
    {
        Happened += handler; //注册一些事件处理方法
    }

    public void RaiseEvent()
    {
        if (Happened != null)
        {
            // Happened(this, null); // 事件永远不会触发，因为此行代码被注释掉了
        }
    }
}
