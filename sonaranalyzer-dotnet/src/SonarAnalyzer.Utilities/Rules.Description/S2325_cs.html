类的方法和属性如果不访问实例的数据就应设为 static 的，不然可能会对方法的使用约定产生误解。

不合规的代码示例

public class Utilities
{
  public int MagicNum // 不合规的
  {
    get
    {
      return 42;
    }
  }

  private static string magicWord = "please";
  public string MagicWord  // 不合规的
  {
    get
    {
      return magicWord;
    }
    set
    {
      magicWord = value;
    }
  }

  public int Sum(int a, int b)  // 不合规的
  {
    return a + b;
  }
}

合规的解决方案

public class Utilities
{
  public static int MagicNum
  {
    get
    {
      return 42;
    }
  }

  private static string magicWord = "please";
  public static string MagicWord
  {
    get
    {
      return magicWord;
    }
    set
    {
      magicWord = value;
    }
  }

  public static int Sum(int a, int b)
  {
    return a + b;
  }
}
