当一次性类型包含以下类型的字段，且未实现终结器时，此规则会引发一个问题：

  • System.IntPtr
  • System.UIntPtr
  • System.Runtime.InteropService.HandleRef

不合规的代码示例

using System;
using System.Runtime.InteropServices;

namespace MyLibrary
{
  public class Foo : IDisposable // 不合规: 没有终结器
  {
    private IntPtr myResource;
    private bool disposed = false;

    protected virtual void Dispose(bool disposing)
    {
      if (!disposed)
      {
        // 处理此实例所持有的资源。
        FreeResource(myResource);
        disposed = true;

        // 禁止已处理实例的终结。
        if (disposing)
        {
          GC.SuppressFinalize(this);
        }
      }
    }

    public void Dispose() {
      Dispose(true);
    }
  }
}

合规的解决方案

using System;
using System.Runtime.InteropServices;

namespace MyLibrary
{
  public class Foo : IDisposable
  {
    private IntPtr myResource;
    private bool disposed = false;

    protected virtual void Dispose(bool disposing)
    {
      if (!disposed)
      {
        // 处理此实例所持有的资源。
        FreeResource(myResource);
        disposed = true;

        // 禁止已处理实例的终结。
        if (disposing)
        {
          GC.SuppressFinalize(this);
        }
      }
    }

    ~Foo()
    {
      Dispose(false);
    }
  }
}
